import pygame import sys import random # Khởi tạo Pygame pygame.init() # Cấu hình màn hình và màu sắc SCREEN_WIDTH, SCREEN_HEIGHT = 500, 650 GRID_SIZE = 8 CELL_SIZE = 40 GRID_OFFSET_X = (SCREEN_WIDTH - (GRID_SIZE * CELL_SIZE)) // 2 GRID_OFFSET_Y = 50 WHITE = (255, 255, 255) BLACK = (20, 20, 20) GRID_COLOR = (40, 40, 40) EMPTY_CELL = (30, 30, 30) SCORE_COLOR = (255, 215, 0) # Danh sách các màu sắc cho khối hình COLORS = [ (255, 50, 50), # Đỏ (50, 255, 50), # Xanh lá (50, 50, 255), # Xanh dương (255, 255, 50), # Vàng (255, 50, 255), # Hồng (50, 255, 255) # Xanh ngọc ] # Định nghĩa các hình dạng khối (Ma trận khối) SHAPES = [ [[1]], # Khối 1 ô [[1, 1]], # Khối ngang 2 ô [[1], [1]], # Khối dọc 2 ô [[1, 1, 1]], # Khối ngang 3 ô [[1, 1], [1, 1]], # Khối vuông 2x2 [[1, 1, 1], [0, 1, 0]], # Khối chữ T [[1, 0], [1, 0], [1, 1]], # Khối chữ L ] screen = pygame.display.set_mode((SCREEN_WIDTH, SCREEN_HEIGHT)) pygame.display.set_caption("Block Blast Mini Python") clock = pygame.time.Clock() font = pygame.font.SysFont("Arial", 30) # Khởi tạo ma trận bàn cờ (0 là trống) grid = [[0 for _ in range(GRID_SIZE)] for _ in range(GRID_SIZE)] score = 0 class Piece: def __init__(self, shape, x, y): self.shape = shape self.color = random.choice(COLORS) self.x = x self.y = y self.is_dragging = False self.original_x = x self.original_y = y self.width = len(shape[0]) * CELL_SIZE self.height = len(shape) * CELL_SIZE def draw(self, surface): for row_idx, row in enumerate(self.shape): for col_idx, val in enumerate(row): if val: pygame.draw.rect(surface, self.color, (self.x + col_idx * CELL_SIZE, self.y + row_idx * CELL_SIZE, CELL_SIZE - 2, CELL_SIZE - 2)) # Tạo 3 khối ngẫu nhiên ở hàng dưới def generate_pieces(): pieces = [] positions = [50, 200, 350] for pos_x in positions: shape = random.choice(SHAPES) pieces.append(Piece(shape, pos_x, 480)) return pieces pieces = generate_pieces() selected_piece = None def check_clear_lines(): global grid, score rows_to_clear = [] cols_to_clear = [] # Check hàng dọc và ngang for r in range(GRID_SIZE): if all(grid[r][c] != 0 for c in range(GRID_SIZE)): rows_to_clear.append(r) for c in range(GRID_SIZE): if all(grid[r][c] != 0 for r in range(GRID_SIZE)): cols_to_clear.append(c) # Xóa và cộng điểm (Tính điểm combo nhẹ) cleared = len(rows_to_clear) + len(cols_to_clear) if cleared > 0: for r in rows_to_clear: for c in range(GRID_SIZE): grid[r][c] = 0 for c in cols_to_clear: for r in range(GRID_SIZE): grid[r][c] = 0 score += cleared * 100 # Vòng lặp game chính running = True while running: screen.fill(BLACK) # Vẽ điểm số score_text = font.render(f"SCORE: {score}", True, SCORE_COLOR) screen.blit(score_text, (20, 10)) # Vẽ lưới game for r in range(GRID_SIZE): for c in range(GRID_SIZE): rect = (GRID_OFFSET_X + c * CELL_SIZE, GRID_OFFSET_Y + r * CELL_SIZE, CELL_SIZE - 1, CELL_SIZE - 1) if grid[r][c] == 0: pygame.draw.rect(screen, EMPTY_CELL, rect) else: pygame.draw.rect(screen, grid[r][c], rect) # Xử lý sự kiện for event in pygame.event.get(): if event.type == pygame.QUIT: running = False elif event.type == pygame.MOUSEBUTTONDOWN: if event.button == 1: # Click chuột trái mx, my = event.pos for p in pieces: if p.x <= mx <= p.x + p.width and p.y <= my <= p.y + p.height: p.is_dragging = True selected_piece = p # Tính độ lệch chuột để kéo mượt hơn offset_x = p.x - mx offset_y = p.y - my break elif event.type == pygame.MOUSEBUTTONUP: if event.button == 1 and selected_piece: selected_piece.is_dragging = False # Tính toán xem khối đang nằm ở ô nào trên lưới grid_c = round((selected_piece.x - GRID_OFFSET_X) / CELL_SIZE) grid_r = round((selected_piece.y - GRID_OFFSET_Y) / CELL_SIZE) # Kiểm tra xem có đặt hợp lệ không can_place = True shape = selected_piece.shape if 0 <= grid_r <= GRID_SIZE - len(shape) and 0 <= grid_c <= GRID_SIZE - len(shape[0]): for r_idx, row in enumerate(shape): for c_idx, val in enumerate(row): if val: if grid[grid_r + r_idx][grid_c + c_idx] != 0: can_place = False else: can_place = False # Nếu đặt được, vẽ vào lưới và xóa khối đó đi if can_place: for r_idx, row in enumerate(shape): for c_idx, val in enumerate(row): if val: grid[grid_r + r_idx][grid_c + c_idx] = selected_piece.color score += 10 pieces.remove(selected_piece) check_clear_lines() else: # Nếu không hợp lệ, trả về vị trí cũ selected_piece.x = selected_piece.original_x selected_piece.y = selected_piece.original_y selected_piece = None # Nếu dùng hết 3 khối, tự đẻ ra 3 khối mới if not pieces: pieces = generate_pieces() elif event.type == pygame.MOUSEMOTION: if selected_piece and selected_piece.is_dragging: mx, my = event.pos selected_piece.x = mx + offset_x selected_piece.y = my + offset_y # Vẽ các khối hình chưa dùng for p in pieces: p.draw(screen) pygame.display.flip() clock.tick(60) pygame.quit() sys.exit()
